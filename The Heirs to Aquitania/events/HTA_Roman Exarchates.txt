#Wins anatolia war, doesnt mean wins the others
country_event = {
	id = 00008
	title = "EVTNAME00008"
	desc = "EVTDESC00008"
	mean_time_to_happen = { days = 7 }
	fire_only_once = yes
	
	trigger = {
		tag = BYZ
		has_global_flag = roman_centralization_crisis
		has_global_flag = anatolia_war_again
		ARM = { NOT = { war_with = THIS } }
		is_disarmed = no
	}
	option = {
		name = "EVTOPTA00008"
		
		prestige = 10
		ARM = {
			set_country_flag = anatolia_war_lost
			any_owned = {
				limit = {
					OR = {
						is_core = BYZ
						region = TUR_894
					}
				}
				add_core = BYZ
				add_core = ANA
				secede_province = BYZ
			}
		}
		
		# If BYZ takes TUR_894 before this event fires, the above any_owned will not add the cores in question
		TUR_894 = {
			add_core = BYZ
			add_core = ANA
		}
	}
}
